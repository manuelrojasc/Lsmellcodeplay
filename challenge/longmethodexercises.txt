import java.util.ArrayList;

class Calidad{

  public void ActualizarCalidad(ArrayList productos){
  	for (var i = 0; i < productos.Count; i++){
  		if (productos[i].nombre != "Queso" && productos[i].nombre != "Pollo"){
  			if (productos[i].calidad > 0){
  				if (productos[i].nombre != "Queso Parmesano"){
  					  productos[i].calidad= productos[i].calidad - 1;
  				}
  			}
  		}else{
  			if (productos[i].calidad < 50){
  				  productos[i].calidad = productos[i].calidad + 1;
  				if (productos[i].Nombre == "pollo"){
  					if (productos[i].preciodeventa < 11){
  						if (productos[i].calidad < 50){
  							productos[i].calidad = productos[i].calidad + 1;
  						}
  					}
  
  					if (productos[i].preciodeventa < 6){
  						if (productos[i].calidad < 50){
  							productos[i].calidad = productos[i].calidad + 1;
  						}
  					}
  				}
  			}
  		}
  
  		if (productos[i].nombre != "Queso Parmesano"){
  			productos[i].preciodeventa = productos[i].preciodeventa - 1;
  		}
  
  		if (productos[i].preciodeventa < 0){
  			if (productos[i].nombre != "Queso"){
  				if (productos[i].Name != "pollo"){
  					if (productos[i].calidad > 0){
  						if (productos[i].Name != "Queso Parmesano"){
  							productos[i].calidad = productos[i].calidad - 1;
  						}
  					}
  				}else{
  					productos[i].calidad = productos[i].calidad - productos[i].calidad;
  				}
  			}else{
  				if (productos[i].calidad < 50){
  					productos[i].calidad = productos[i].calidad + 1;
  				}
  			}
  		}
  	}
  } 
}
